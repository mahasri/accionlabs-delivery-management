# Friday Learning Hour

Friday Learnings is our weekly knowledge sharing platform where leaders share insights, best practices, and key takeaways from their participation in various training programs, certifications, workshops, and industry summits. This program ensures that valuable knowledge gained from external learning experiences is disseminated across the organization, maximizing the return on investment in leadership development.

On average, every leader participates in 2-3 deep training programs and workshops annually, bringing back cutting-edge methodologies, industry trends, and innovative approaches that directly benefit our delivery teams and client projects.

<script>
let flSlideIndex = 1;
showFlSlides(flSlideIndex);

function changeFlSlide(n) {
    showFlSlides(flSlideIndex += n);
}

function currentFlSlide(n) {
    showFlSlides(flSlideIndex = n);
}

function showFlSlides(n) {
    let i;
    let slides = document.getElementsByClassName("fl-slide");
    let dots = document.getElementsByClassName("fl-dot");
    
    if (n > slides.length) {flSlideIndex = 1}
    if (n < 1) {flSlideIndex = slides.length}
    
    for (i = 0; i < slides.length; i++) {
        slides[i].style.display = "none";
    }
    for (i = 0; i < dots.length; i++) {
        dots[i].className = dots[i].className.replace(" active", "");
    }
    
    slides[flSlideIndex-1].style.display = "block";
    dots[flSlideIndex-1].className += " active";
}

// Auto advance every 3 seconds
setInterval(function() {
    changeFlSlide(1);
}, 3000);
</script>
